DWORD Decrypt::CIndex(DWORD value)
{
    DWORD ecx = value ^ 0x1EDBF73C;
    DWORD edx = ecx;
    edx = _rotr(edx, 0x1F);
    DWORD eax = ecx;
    eax = eax >> 0x0F;
    eax = eax & 0x00010000;
    ecx = ecx << 0x11;
    eax = eax | ecx;
    eax = eax ^ 0xFCA25AFA;
    edx = edx ^ eax;
    return edx;
}
